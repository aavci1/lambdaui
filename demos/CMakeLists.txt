function(lambdaui_add_demo target_name)
  string(REPLACE "-" "." _lambda_demo_suffix "${target_name}")
  lambdaui_add_app(${target_name}
    APP_NAME "${target_name}"
    BUNDLE_IDENTIFIER "io.lambda.demos.${_lambda_demo_suffix}"
    COMMENT "Lambda demo"
    SOURCES ${ARGN})
endfunction()
